
_printRAX:
    mov rcx, digitSpace ; beginning of the string
    mov rbx, 10 ; add a newline character
    mov [rcx], rbx
    inc rcx
    mov [digitSpacePos], rcx

_printRAXLoop:
    mov rdx, 0
    mov rbx, 10
    div rbx ; divide it by 10
    push rax
    add rdx, 48 ; convert it to an ASCII char

    mov rcx, [digitSpacePos]
    mov [rcx], dl ; loads the ASCII char
    inc rcx
    mov [digitSpacePos], rcx

    pop rax,
    cmp rax, 0
    jne _printRAXLoop

_printRAXLoop2:
    mov rcx, [digitSpacePos]
    mov rax, 1
    mov rdi, 1
    mov rsi, rcx
    mov rdx, 1
    syscall

    mov rcx, [digitSpacePos]
    dec rcx,
    mov [digitSpacePos], rcx

    cmp rcx, digitSpace
    jge _printRAXLoop2
    ret
